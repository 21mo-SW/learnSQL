Oracl SQL Developer 스튜디오에 SYSTEM 사용자로 로그인 후
아래 sql문 실행하여 id/비밀번호가 flightinfo/pass1234인
사용자를 생성한다.

alter session set "_ORACLE_SCRIPT"=true;
CREATE USER flightinfo IDENTIFIED BY pass1234;
alter session set "_ORACLE_SCRIPT"=false;

GRANT CONNECT,RESOURCE,UNLIMITED TABLESPACE, DBA TO flightinfo;
GRANT CREATE SESSION TO flightinfo;
ALTER USER flightinfo DEFAULT TABLESPACE USERS;
ALTER USER flightinfo TEMPORARY TABLESPACE TEMP;


위에서 생성한 id/비밀번호로 접속하여
필요한 테이블을 만들어야 한다.


CREATE TABLE AIRLINE (
	ID NUMBER PRIMARY KEY,
	NAME VARCHAR2(30) NOT NULL,
	ABBR VARCHAR2(3) NOT NULL);

CREATE TABLE AIRPORT (
	ID NUMBER PRIMARY KEY,
	LOC VARCHAR2(15) NOT NULL,
	CODE VARCHAR2(7) NOT NULL);

CREATE TABLE USERS (
	ID NUMBER PRIMARY KEY,
	USERID VARCHAR2(30) NOT NULL,
	PASSWORD VARCHAR2(30) NOT NULL,
	PHONE VARCHAR2(11) NOT NULL,
	ADMIN CHAR(1) NOT NULL);

CREATE TABLE SCHEDULE (
	ID NUMBER PRIMARY KEY,
	AIRLINE_ID NUMBER NOT NULL,
	DEPARTLOCID NUMBER NOT NULL,
	ARRIVALLOCID NUMBER NOT NULL,
	DEPARTTIME NUMBER NOT NULL,
	ARRIVALTIME NUMBER NOT NULL,
	ECONOMY NUMBER NOT NULL,
	PRESTIGE NUMBER NOT NULL,
	VEHICLEID VARCHAR(8) NOT NULL
);


CREATE TABLE SCHEDULE (
	ID NUMBER PRIMARY KEY,
	AIRLINENAME VARCHAR2(30) NOT NULL,
	DEPARTLOCNAME VARCHAR(15) NOT NULL,
	ARRIVALLOCNAME VARCHAR(15) NOT NULL,
	DEPARTTIME NUMBER NOT NULL,
	ARRIVALTIME NUMBER NOT NULL,
	ECONOMY NUMBER,
	PRESTIGE NUMBER,
	VIHICLEID VARCHAR(8) NOT NULL
);
